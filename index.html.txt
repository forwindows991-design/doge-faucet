    <script>
        // Simple JavaScript for the Timer and Button State (Not persistent, but functional)
        const timerElement = document.getElementById('timer');
        const claimButton = document.getElementById('claimButton');
        const claimForm = document.getElementById('claimForm');
        let countdownSeconds = 0; // Starts at 0:00 as per image

        function updateTimerDisplay() {
            const minutes = Math.floor(countdownSeconds / 60);
            const seconds = countdownSeconds % 60;
            timerElement.textContent = `${minutes}:${seconds.toString().padStart(2, '0')}`;

            if (countdownSeconds > 0) {
                claimButton.disabled = true;
                claimButton.classList.replace('bg-doge-orange', 'bg-gray-400');
                claimButton.classList.replace('hover:bg-doge-dark', 'hover:bg-gray-400');
            } else {
                claimButton.disabled = false;
                claimButton.classList.replace('bg-gray-400', 'bg-doge-orange');
                claimButton.classList.replace('hover:bg-gray-400', 'hover:bg-doge-dark');
            }
        }

        function startCountdown() {
            updateTimerDisplay(); // Update immediately on load

            const interval = setInterval(() => {
                if (countdownSeconds > 0) {
                    countdownSeconds--;
                    updateTimerDisplay();
                } else {
                    // Timer reached 0:00, stop countdown and enable button
                    clearInterval(interval);
                    updateTimerDisplay(); 
                    console.log('Claim ready!');
                }
            }, 1000);
        }

        claimForm.addEventListener('submit', function(e) {
            e.preventDefault();
            
            // In a real application, you would send the wallet address to a server here.
            console.log("Claim Attempted for Address: ", document.getElementById('walletAddress').value);

            // Simulate setting a new claim timer (5 minutes - UPDATED)
            countdownSeconds = 5 * 60; 
            
            // Restart the countdown
            startCountdown();

            // Clear input for aesthetic purposes
            document.getElementById('walletAddress').value = '';

            // Simple success message (replace with a custom modal in a real app)
            console.log("Claim initiated! Please wait for the timer to finish.");
        });

        // Initialize the timer on window load
        window.onload = startCountdown;
    </script>
